001package fr.aumgn.bukkitutils.itemtype;
002
003public class MonsterEggsDataParser extends ItemTypeDataParser {
004
005    @Override
006    public Short parse(String token) {
007        if (token.equalsIgnoreCase("stone")) {
008            return 0;
009        }
010        if (token.equalsIgnoreCase("cobblestone")) {
011            return 1;
012        }
013        if (token.equalsIgnoreCase("smooth_brick")) {
014            return 2;
015        }
016
017        return null;
018    }
019}
