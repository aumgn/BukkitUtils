001package fr.aumgn.bukkitutils.playerref.map;
002
003import java.util.Map.Entry;
004
005import org.bukkit.entity.Player;
006
007import fr.aumgn.bukkitutils.playerref.PlayerRef;
008
009public class PlayerMapEntry<T> implements Entry<Player, T> {
010
011    private final Entry<PlayerRef, T> entry;
012
013    public PlayerMapEntry(Entry<PlayerRef, T> entry) {
014        this.entry = entry;
015    }
016
017    @Override
018    public Player getKey() {
019        return entry.getKey().getPlayer();
020    }
021
022    @Override
023    public T getValue() {
024        return entry.getValue();
025    }
026
027    @Override
028    public T setValue(T value) {
029        return entry.setValue(value);
030    }
031}
