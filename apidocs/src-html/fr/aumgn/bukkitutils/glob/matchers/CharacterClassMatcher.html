001package fr.aumgn.bukkitutils.glob.matchers;
002
003import org.apache.commons.lang.CharSet;
004
005public class CharacterClassMatcher implements GlobMatcher {
006
007    private final CharSet charSet;
008
009    public CharacterClassMatcher(String string) {
010        charSet = CharSet.getInstance(string);
011    }
012
013    public CharSet getCharSet() {
014        return charSet;
015    }
016
017    @Override
018    public int match(String str, int from) {
019        if (from >= str.length()) {
020            return -1;
021        }
022
023        char character = str.charAt(from);
024        if (charSet.contains(character)) {
025            return from + 1;
026        }
027
028        return -1;
029    }
030}
