001package fr.aumgn.bukkitutils.glob.matchers;
002
003import java.util.Collections;
004import java.util.List;
005
006
007public class WildcardMatcher implements GlobMatcher {
008
009    private final List<GlobMatcher> matchers;
010
011    public WildcardMatcher(String pattern, int i) {
012        // "**" is strictly equivalent to "*"
013        // So forward to first non wildcard character
014        // to avoid unnecessary calculation.
015        while (i < pattern.length() && pattern.charAt(i) == '*') {
016            i++;
017        }
018
019        this.matchers = Collections.unmodifiableList(
020                GlobParser.parse(pattern, i));
021    }
022
023    public List<GlobMatcher> getMatchers() {
024        return matchers;
025    }
026
027    @Override
028    public int match(String str, int from) {
029        if (matchers.isEmpty()) {
030            return str.length();
031        }
032
033        wildcard:
034        for (int i = from; i < str.length(); i++) {
035            int index = i;
036            for (GlobMatcher matcher : matchers) {
037                index = matcher.match(str, index);
038                if (index < 0) {
039                    continue wildcard;
040                }
041            }
042            if (index == str.length()) {
043                return index;
044            }
045        }
046
047        return -1;
048    }
049}
