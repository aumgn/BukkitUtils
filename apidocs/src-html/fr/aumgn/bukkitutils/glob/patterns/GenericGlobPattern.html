001package fr.aumgn.bukkitutils.glob.patterns;
002
003import com.google.common.base.Function;
004
005public class GenericGlobPattern<T> extends AbstractGlobPattern<T> {
006
007    private final Function<T, String> toString;
008
009    public GenericGlobPattern(String pattern, Function<T, String> toString) {
010        super(pattern);
011        this.toString = toString;
012    }
013
014    public boolean match(T obj) {
015        return match(toString.apply(obj));
016    }
017}
