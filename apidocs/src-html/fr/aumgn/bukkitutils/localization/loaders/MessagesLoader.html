001package fr.aumgn.bukkitutils.localization.loaders;
002
003import java.io.Reader;
004import java.text.MessageFormat;
005import java.util.HashMap;
006import java.util.Locale;
007import java.util.Map;
008import java.util.Map.Entry;
009
010import fr.aumgn.bukkitutils.util.Util;
011
012/**
013 * Interfaces for classes which load messages
014 * for a specific format.
015 */
016public abstract class MessagesLoader {
017
018    /**
019     * File extensions (without the dot) accepted
020     * by this loader.
021     */
022    public abstract String[] getExtensions();
023
024    /**
025     * Parse the reader as a Map of &lt;key, message&gt;.
026     */
027    public abstract Map<?, ?> loadRaw(Reader reader);
028
029    public Map<String, MessageFormat> load(
030            Locale locale, Reader reader) {
031        Map<?, ?> loaded = loadRaw(reader);
032        Map<String, MessageFormat> map = new HashMap<String, MessageFormat>();
033
034        for (Entry<?, ?> entry : loaded.entrySet()) {
035            String key = entry.getKey().toString();
036            String message = entry.getValue().toString();
037            map.put(key, parse(locale, message));
038        }
039
040        return map;
041    }
042
043    protected MessageFormat parse(Locale locale, String rawMessage) {
044        String message = Util.parseColorsMarkup(rawMessage);
045        message = message.replaceAll("'", "''");
046        message = message.replaceAll("\\\\", "'");
047        return new MessageFormat(message, locale);
048    }
049}
