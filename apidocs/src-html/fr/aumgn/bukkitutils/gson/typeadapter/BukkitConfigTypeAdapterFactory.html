001package fr.aumgn.bukkitutils.gson.typeadapter;
002
003import java.io.IOException;
004import java.lang.reflect.Type;
005import java.util.HashMap;
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009import org.bukkit.configuration.serialization.ConfigurationSerializable;
010import org.bukkit.configuration.serialization.ConfigurationSerialization;
011
012import com.google.gson.Gson;
013import com.google.gson.TypeAdapter;
014import com.google.gson.TypeAdapterFactory;
015import com.google.gson.reflect.TypeToken;
016import com.google.gson.stream.JsonReader;
017import com.google.gson.stream.JsonWriter;
018
019/**
020 * Serialize with gson classes which implements
021 * Bukkit's {@link ConfigurationSerialization}.
022 */
023@SuppressWarnings("unchecked")
024public class BukkitConfigTypeAdapterFactory implements TypeAdapterFactory {
025
026    public static class BukkitConfigTypeAdapter<T> extends TypeAdapter<T> {
027
028        private static final String SERIALIZED_TYPE_KEY = "$$";
029        private static final Type BUKKIT_SERIAL_TYPE =
030                new TypeToken<HashMap<String, String>>() {}.getType();
031
032        private final Gson gson;
033
034        public BukkitConfigTypeAdapter(Gson gson) {
035            this.gson = gson;
036        }
037
038        @Override
039        public T read(JsonReader reader) throws IOException {
040            Map<String, Object> map = gson.fromJson(
041                    reader, BUKKIT_SERIAL_TYPE);
042            map.put(ConfigurationSerialization.SERIALIZED_TYPE_KEY,
043                    map.remove(SERIALIZED_TYPE_KEY));
044
045            return (T) ConfigurationSerialization.deserializeObject(map);
046        }
047
048        @Override
049        public void write(JsonWriter writer, T obj) throws IOException {
050            Map<String, Object> map = new LinkedHashMap<String, Object>();
051            ConfigurationSerializable serializable =
052                    (ConfigurationSerializable) obj;
053            map.put(SERIALIZED_TYPE_KEY,
054                    ConfigurationSerialization.getAlias(
055                            serializable.getClass()));
056            map.putAll(serializable.serialize());
057
058            gson.toJson(map, BUKKIT_SERIAL_TYPE, writer);
059        }
060    }
061
062    @Override
063    public <T> TypeAdapter<T> create(Gson gson, TypeToken<T> token) {
064        if (!ConfigurationSerializable.class.isAssignableFrom(
065                token.getRawType())) {
066            return null;
067        }
068
069        return new BukkitConfigTypeAdapter<T>(gson);
070    }
071}
