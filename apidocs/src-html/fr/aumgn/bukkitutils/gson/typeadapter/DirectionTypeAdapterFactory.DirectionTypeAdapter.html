001package fr.aumgn.bukkitutils.gson.typeadapter;
002
003import java.io.IOException;
004
005import com.google.gson.Gson;
006import com.google.gson.TypeAdapter;
007import com.google.gson.TypeAdapterFactory;
008import com.google.gson.reflect.TypeToken;
009import com.google.gson.stream.JsonReader;
010import com.google.gson.stream.JsonToken;
011import com.google.gson.stream.JsonWriter;
012
013import fr.aumgn.bukkitutils.geom.Direction;
014import fr.aumgn.bukkitutils.geom.Directions;
015
016public class DirectionTypeAdapterFactory implements TypeAdapterFactory {
017
018    public static class DirectionTypeAdapter<T> extends TypeAdapter<T> {
019
020        @Override
021        public void write(JsonWriter writer, T obj) throws IOException {
022            if (obj == Direction.NONE) {
023                writer.value("NONE");
024                return;
025            }
026
027            Direction dir = (Direction) obj;
028            writer.beginObject();
029            writer.name("yaw");
030            writer.value(dir.getYaw());
031            writer.name("pitch");
032            writer.value(dir.getPitch());
033            writer.endObject();
034        }
035
036        @SuppressWarnings("unchecked")
037        @Override
038        public T read(JsonReader reader) throws IOException {
039            if (reader.peek() == JsonToken.STRING
040                    && reader.nextString().equals("NONE")) {
041                return (T) Direction.NONE;
042            }
043
044            reader.beginObject();
045            reader.nextName();
046            float yaw = (float) reader.nextDouble();
047            reader.nextName();
048            float pitch = (float) reader.nextDouble();
049            reader.endObject();
050            return (T) Directions.fromYawAndPitch(yaw, pitch);
051        }
052    }
053
054    @Override
055    public <T> TypeAdapter<T> create(Gson gson, TypeToken<T> type) {
056        if (!Direction.class.isAssignableFrom(type.getRawType())) {
057            return null;
058        }
059
060        return new DirectionTypeAdapter<T>();
061    }
062}
