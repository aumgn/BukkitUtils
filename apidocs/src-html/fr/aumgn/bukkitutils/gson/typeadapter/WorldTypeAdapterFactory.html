001package fr.aumgn.bukkitutils.gson.typeadapter;
002
003import java.io.IOException;
004import java.util.UUID;
005
006import org.bukkit.Bukkit;
007import org.bukkit.World;
008
009import com.google.gson.Gson;
010import com.google.gson.TypeAdapter;
011import com.google.gson.TypeAdapterFactory;
012import com.google.gson.reflect.TypeToken;
013import com.google.gson.stream.JsonReader;
014import com.google.gson.stream.JsonWriter;
015
016import fr.aumgn.bukkitutils.gson.GsonLoadRuntimeException;
017
018@SuppressWarnings("unchecked")
019public class WorldTypeAdapterFactory implements TypeAdapterFactory {
020
021    private final boolean throwsException;
022
023    public WorldTypeAdapterFactory() {
024        this(true);
025    }
026
027    public WorldTypeAdapterFactory(boolean throwsException) {
028        this.throwsException = throwsException;
029    }
030
031    @Override
032    public <T> TypeAdapter<T> create(final Gson gson, TypeToken<T> token) {
033        if (!World.class.isAssignableFrom(token.getRawType())) {
034            return null;
035        }
036
037        return new TypeAdapter<T>() {
038
039            @Override
040            public T read(JsonReader reader) throws IOException {
041                UUID uuid = gson.fromJson(reader, UUID.class);
042                World world = Bukkit.getWorld(uuid);
043                if (world == null && throwsException) {
044                    throw new GsonLoadRuntimeException(
045                            "Unable to load world for uuid :" + uuid);
046                }
047
048                return (T) world;
049            }
050
051            @Override
052            public void write(JsonWriter writer, T obj) throws IOException {
053                World world = (World) obj;
054                gson.toJson(world.getUID(), UUID.class, writer);
055            }
056        };
057    }
058}
