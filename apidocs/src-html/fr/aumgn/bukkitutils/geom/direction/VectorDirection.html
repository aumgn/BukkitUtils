001package fr.aumgn.bukkitutils.geom.direction;
002
003import static com.google.common.base.Preconditions.checkArgument;
004import static fr.aumgn.bukkitutils.geom.direction.DirectionUtil.calculatePitch;
005import static fr.aumgn.bukkitutils.geom.direction.DirectionUtil.calculateYaw;
006import fr.aumgn.bukkitutils.geom.Direction;
007import fr.aumgn.bukkitutils.geom.Vector;
008import fr.aumgn.bukkitutils.geom.Vector2D;
009
010public final class VectorDirection extends AbstractDirection
011        implements Direction {
012
013    private final Vector vector;
014    private float yaw;
015    private float pitch;
016
017    public VectorDirection(Vector vector) {
018        checkArgument(!vector.isZero(),
019                "Cannot create a direction with a null vector.");
020        this.vector = vector.normalize();
021        this.yaw = Float.NaN;
022        this.pitch = Float.NaN;
023    }
024
025    @Override
026    public float getYaw() {
027        if (Float.isNaN(yaw)) {
028            yaw = calculateYaw(vector.to2D());
029        }
030
031        return yaw;
032    }
033
034    @Override
035    public float getPitch() {
036        if (Float.isNaN(pitch)) {
037            pitch = calculatePitch(vector);
038        }
039
040        return pitch;
041    }
042
043    @Override
044    public Vector2D getVector2D() {
045        return vector.to2D();
046    }
047
048    @Override
049    public Vector getVector() {
050        return vector;
051    }
052
053    @Override
054    public boolean equals(Object obj) {
055        if (obj instanceof VectorDirection) {
056            return vector.equals(((VectorDirection) obj).vector);
057        }
058
059        return super.equals(obj);
060    }
061}
