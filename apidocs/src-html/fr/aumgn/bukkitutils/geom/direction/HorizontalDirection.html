001package fr.aumgn.bukkitutils.geom.direction;
002
003import fr.aumgn.bukkitutils.geom.Direction;
004import fr.aumgn.bukkitutils.geom.Vector;
005import fr.aumgn.bukkitutils.geom.Vector2D;
006import static fr.aumgn.bukkitutils.geom.direction.DirectionUtil.*;
007
008public final class HorizontalDirection extends AbstractDirection {
009
010    protected final float yaw;
011    private Vector2D vector;
012
013    public HorizontalDirection(float yaw) {
014        this.yaw = yaw % 360;
015        this.vector = null;
016    }
017
018    public float getYaw() {
019        return yaw;
020    }
021
022    @Override
023    public float getPitch() {
024        return 0.0f;
025    }
026
027    public Vector2D getVector2D() {
028        if (vector == null) {
029            vector = calculateVector2D(yaw);
030        }
031
032        return vector;
033    }
034
035    @Override
036    public Vector getVector() {
037        return getVector2D().to3D();
038    }
039
040    @Override
041    public Direction rotate(float angle) {
042        return calculateHorizontalRotation(this, angle);
043    }
044}
