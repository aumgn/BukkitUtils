001package fr.aumgn.bukkitutils.geom.direction;
002
003import static com.google.common.base.Preconditions.checkArgument;
004import static fr.aumgn.bukkitutils.geom.direction.DirectionUtil.calculateHorizontalRotation;
005import static fr.aumgn.bukkitutils.geom.direction.DirectionUtil.calculateYaw;
006import fr.aumgn.bukkitutils.geom.Direction;
007import fr.aumgn.bukkitutils.geom.Vector;
008import fr.aumgn.bukkitutils.geom.Vector2D;
009
010public final class Vector2DDirection extends AbstractDirection
011        implements Direction {
012
013    private final Vector2D vector;
014    private float yaw;
015
016    public Vector2DDirection(Vector2D vector) {
017        checkArgument(!vector.isZero(),
018                "Cannot create a direction with a null vector.");
019        this.vector = vector.normalize();
020        this.yaw = Float.NaN;
021    }
022
023    @Override
024    public float getYaw() {
025        if (Float.isNaN(yaw)) {
026            yaw = calculateYaw(vector);
027        }
028
029        return yaw;
030    }
031
032    @Override
033    public float getPitch() {
034        return 0.0f;
035    }
036
037    @Override
038    public Vector2D getVector2D() {
039        return vector;
040    }
041
042    @Override
043    public Vector getVector() {
044        return vector.to3D();
045    }
046
047    @Override
048    public Direction rotate(float angle) {
049        return calculateHorizontalRotation(this, angle);
050    }
051
052    @Override
053    public boolean equals(Object obj) {
054        if (obj instanceof Vector2DDirection) {
055            return vector.equals(((Vector2DDirection) obj).vector);
056        }
057
058        return super.equals(obj);
059    }
060}
