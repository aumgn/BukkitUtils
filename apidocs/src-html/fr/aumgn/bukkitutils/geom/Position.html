001package fr.aumgn.bukkitutils.geom;
002
003import java.lang.ref.WeakReference;
004import java.util.UUID;
005
006import org.bukkit.Bukkit;
007import org.bukkit.Location;
008import org.bukkit.World;
009
010/**
011 * Store informations of a position.
012 *
013 * Informations are :
014 * <ul>
015 *  <li>World</li>
016 *  <li>Coordinates</li>
017 *  <li>Direction</li>
018 * </ul>
019 *
020 * Contrary to {@link Location}, this class stores a
021 * reference to the world rather than the world itself.
022 */
023public class Position {
024
025    private UUID worldId;
026    private Vector coord;
027    private Direction direction;
028
029    private transient WeakReference<World> worldRef = null;
030
031    public Position(Location location) {
032        setWorld(location.getWorld());
033        setCoord(new Vector(location));
034        setDirection(Directions.fromLocation(location));
035    }
036
037    public World getWorld() {
038        World world = worldRef.get();
039        if (world == null) {
040            world = Bukkit.getWorld(worldId);
041            worldRef = new WeakReference<World>(world);
042        }
043
044        return world;
045    }
046
047    public void setWorld(World world) {
048        worldId = world.getUID();
049        worldRef = new WeakReference<World>(world);
050    }
051
052    public Vector getCoord() {
053        return coord;
054    }
055
056    private void setCoord(Vector vector) {
057        coord = vector;
058    }
059
060    public Direction getDirection() {
061        return direction;
062    }
063
064    public void setDirection(Direction dir) {
065        direction = dir;
066    }
067
068    public Location toLocation() {
069        return coord.toLocation(getWorld(), direction);
070    }
071}
