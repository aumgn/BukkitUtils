001package fr.aumgn.bukkitutils.command.executor;
002
003import java.lang.reflect.InvocationTargetException;
004import java.lang.reflect.Method;
005import java.util.HashSet;
006import java.util.Set;
007import java.util.logging.Level;
008
009import org.bukkit.Bukkit;
010import org.bukkit.ChatColor;
011import org.bukkit.command.CommandExecutor;
012import org.bukkit.command.CommandSender;
013import org.bukkit.entity.Player;
014
015import fr.aumgn.bukkitutils.command.Command;
016import fr.aumgn.bukkitutils.command.Commands;
017import fr.aumgn.bukkitutils.command.CommandsMessages;
018import fr.aumgn.bukkitutils.command.args.CommandArgs;
019import fr.aumgn.bukkitutils.command.args.CommandArgsParser;
020import fr.aumgn.bukkitutils.command.exception.CommandException;
021import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
022import fr.aumgn.bukkitutils.glob.exceptions.GlobException;
023import fr.aumgn.bukkitutils.glob.exceptions.UnbalancedSquareBracketException;
024
025public class MethodCommandExecutor implements CommandExecutor {
026
027    private final CommandsMessages messages;
028    private final Commands instance;
029    private final Method preExecute;
030    private final Method method;
031    private final int min;
032    private final int max;
033    private final Set<Character> flags;
034    private final Set<Character> argsFlags;
035    private final boolean isPlayerCommand;
036
037    public MethodCommandExecutor(CommandsMessages messages, Commands instance,
038            Method preExecute, Method method, Command command) {
039        this.messages = messages;
040        this.instance = instance;
041        this.preExecute = preExecute;
042        this.method = method;
043
044        if (method.getParameterTypes().length > 1) {
045            this.min = command.min();
046            this.max = command.max();
047        } else {
048            this.min = -1;
049            this.max = 0;
050        }
051        this.flags = new HashSet<Character>();
052        for (char flag : command.flags().toCharArray()) {
053            this.flags.add(flag);
054        }
055        this.argsFlags = new HashSet<Character>();
056        for (char flag : command.argsFlags().toCharArray()) {
057            this.argsFlags.add(flag);
058        }
059        this.isPlayerCommand = Player.class.isAssignableFrom(
060                method.getParameterTypes()[0]);
061    }
062
063    @Override
064    public boolean onCommand(CommandSender sender,
065            org.bukkit.command.Command cmd, String lbl, String[] rawArgs) {
066        if (isPlayerCommand && !(sender instanceof Player)) {
067            sender.sendMessage(ChatColor.RED + messages.playerOnly());
068            return true;
069        }
070        try {
071            CommandArgs args;
072            if (min >= 0) {
073                args = getArgs(rawArgs);
074            } else {
075                if (rawArgs.length > 0) {
076                    throw new CommandUsageError(
077                            messages.tooManyArguments(rawArgs.length, 0));
078                }
079                args = null;
080            }
081            callCommand(lbl, sender, args);
082        } catch (GlobException exc) {
083            handleGlobException(exc);
084        } catch (CommandUsageError error) {
085            sender.sendMessage(ChatColor.RED + error.getMessage());
086            return false;
087        } catch (Throwable thr) {
088            sender.sendMessage(ChatColor.RED + thr.getMessage());
089        }
090        return true;
091    }
092
093    private CommandArgs getArgs(String[] rawArgs){
094        CommandArgsParser parser = new CommandArgsParser(messages, rawArgs);
095        parser.validate(flags, argsFlags, min, max);
096        return new CommandArgs(messages, parser);
097    }
098
099    private void callCommand(String name, CommandSender sender,
100            CommandArgs args) throws Throwable {
101        try {
102            if (preExecute != null) {
103                preExecute.invoke(instance, sender, args);
104            }
105            if (args != null) {
106                method.invoke(instance, sender, args);
107            } else {
108                method.invoke(instance, sender);
109            }
110        } catch (InvocationTargetException exc) {
111            Throwable cause = exc.getCause();
112            if (cause instanceof CommandException) {
113                throw cause;
114            }
115            unhandledError(name, args, cause);
116        } catch (IllegalArgumentException exc) {
117            unhandledError(name, args, exc);
118        } catch (IllegalAccessException exc) {
119            unhandledError(name, args, exc);
120        }
121    }
122
123    private void handleGlobException(GlobException exc) {
124        if (exc instanceof UnbalancedSquareBracketException) {
125            UnbalancedSquareBracketException usbExc;
126            usbExc = ((UnbalancedSquareBracketException) exc);
127            throw new CommandUsageError(
128                    messages.globUnbalancedSquareBracket(usbExc.getGlob()));
129        }
130
131        throw new RuntimeException(exc);
132    }
133
134    private void unhandledError(String name, CommandArgs args, Throwable exc) {
135        if (!(exc instanceof org.bukkit.command.CommandException)) {
136            Bukkit.getLogger().severe(
137                    "Exception occured while executing \""+ name + "\"");
138            if (args != null) {
139                if (args.hasFlags()) {
140                    StringBuilder flagsString = new StringBuilder();
141                    for (char flag : args.flags()) {
142                        flagsString.append(flag);
143                    }
144                    Bukkit.getLogger().severe(
145                            "Flags : " + flagsString.toString());
146                }
147                if (args.length() > 0) {
148                    StringBuilder arguments = new StringBuilder();
149                    for (String arg : args.asList()) {
150                        arguments.append(arg);
151                        arguments.append(" ");
152                    }
153                    Bukkit.getLogger().severe(
154                            "Arguments : " + arguments.toString());
155                }
156            }
157        }
158        Bukkit.getLogger().log(Level.SEVERE, "Exception : ", exc);
159    }
160}
