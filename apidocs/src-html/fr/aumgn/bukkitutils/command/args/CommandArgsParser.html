001package fr.aumgn.bukkitutils.command.args;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.HashSet;
006import java.util.Map;
007import java.util.Set;
008
009import fr.aumgn.bukkitutils.command.CommandsMessages;
010import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
011
012public class CommandArgsParser {
013
014    private CommandsMessages messages;
015    private String[] args;
016    private Set<Character> flags;
017    private Map<Character, String> argsFlags;
018
019    public CommandArgsParser(CommandsMessages messages, String[] tokens) {
020        this.messages = messages;
021        parse(tokens);
022    }
023
024    private void parse(String[] tokens) {
025        flags = new HashSet<Character>();
026        argsFlags = new HashMap<Character, String>();
027        ArrayList<String> argsList = new ArrayList<String>();
028        boolean quoted = false;
029        StringBuilder current = null;
030
031        for (String token : tokens) {
032
033            if (quoted) {
034                if (!token.isEmpty()
035                        && token.charAt(token.length() - 1) == '\"') {
036                    current.append(" ");
037                    current.append(token.substring(0, token.length() - 1));
038                    argsList.add(current.toString());
039                    quoted = false;
040                } else {
041                    current.append(" ");
042                    current.append(token);
043                }
044            } else {
045                if (token.isEmpty()) {
046                    continue;
047                }
048
049                if (token.charAt(0) == '"') {
050                    if (token.charAt(token.length() - 1) == '\"') {
051                        argsList.add(token.substring(1, token.length() - 1));
052                    } else {
053                        quoted = true;
054                        current = new StringBuilder();
055                        current.append(token.substring(1));
056                    }
057                } else if (token.charAt(0) == '-' && token.length() > 1
058                        && Character.isLetter(token.charAt(1))) {
059                    parseFlags(token.substring(1));
060                } else {
061                    argsList.add(token);
062                }
063            }
064        }
065
066        if (quoted) {
067            throw new CommandUsageError(
068                    messages.missingEndingQuote(current.toString()));
069        }
070
071        args = argsList.toArray(new String[argsList.size()]);
072    }
073
074    private void parseFlags(String flagsString) {
075        int equal = flagsString.indexOf("=");
076        if (equal == -1) {
077            parseRegularFlags(flagsString);
078            return;
079        }
080
081        if (equal > 1) {
082            parseRegularFlags(flagsString.substring(0, equal -1));
083        }
084
085        argsFlags.put(
086                flagsString.charAt(equal - 1),
087                flagsString.substring(equal + 1));
088    }
089
090    private void parseRegularFlags(String flagsString) {
091        for (char flag : flagsString.toCharArray()) {
092            flags.add(flag);
093        }
094    }
095
096    public void validate(Set<Character> allowedFlags,
097            Set<Character> allowedArgsFlags, int min, int max) {
098        StringBuilder invalidFlags = new StringBuilder();
099        for (char flag : flags) {
100            if (!allowedFlags.contains(flag)) {
101                invalidFlags.append(flag);
102            }
103        }
104
105        for (char flag : argsFlags.keySet()) {
106            if (!allowedArgsFlags.contains(flag)) {
107                invalidFlags.append(flag);
108            }
109        }
110
111        if (invalidFlags.length() > 0) {
112            throw new CommandUsageError(
113                    messages.invalidFlag(invalidFlags.toString()));
114        }
115
116        if (args.length < min) {
117            throw new CommandUsageError(
118                    messages.missingArguments(args.length, min));
119        }
120        if (max != -1 && args.length > max) {
121            throw new CommandUsageError(
122                    messages.tooManyArguments(args.length, max));
123        }
124    }
125
126    public Set<Character> getFlags() {
127        return flags;
128    }
129
130    public Map<Character, String> getArgsFlags() {
131        return argsFlags;
132    }
133
134    public String[] getArgs() {
135        return args;
136    }
137}
