001package fr.aumgn.bukkitutils.command.exception;

003/**
004 * Thrown when an error has been detected in the
005 * usage of the command.
006 */
007public class CommandUsageError extends RuntimeException
008        implements CommandException {
009    private static final long serialVersionUID = -2455651120183829047L;
010
011    public CommandUsageError() {
012        super();
013    }
014
015    public CommandUsageError(String message) {
016        super(message);
017    }
018}
