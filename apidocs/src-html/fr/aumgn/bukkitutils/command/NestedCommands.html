001package fr.aumgn.bukkitutils.command;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * Defines a set of nested commands.
010 *
011 * All {@link Commands} subclasses which has this
012 * annotation will prepend the given prefix to
013 * their commands when registered. 
014 */
015@Target(ElementType.TYPE)
016@Retention(RetentionPolicy.RUNTIME)
017public @interface NestedCommands {
018
019    String[] value();
020}
