001package fr.aumgn.bukkitutils.command.arg.basic;
002
003import fr.aumgn.bukkitutils.command.CommandsMessages;
004import fr.aumgn.bukkitutils.command.arg.impl.AbstractCommandArg;
005import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
006
007public class DoubleArg extends AbstractCommandArg<Double> {
008
009    public DoubleArg(CommandsMessages messages, String string) {
010        super(messages, string);
011    }
012
013    @Override
014    public Double value() {
015        try {
016            return Double.parseDouble(string);
017        } catch (NumberFormatException exc) {
018            throw new CommandUsageError(messages.notAValidNumber(string));
019        }
020    }
021}
