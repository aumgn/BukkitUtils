001package fr.aumgn.bukkitutils.command.arg.basic;
002
003import org.bukkit.command.CommandSender;
004import org.bukkit.entity.Player;
005
006import fr.aumgn.bukkitutils.command.CommandsMessages;
007import fr.aumgn.bukkitutils.command.arg.impl.AsbtractSenderArg;
008import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
009import fr.aumgn.bukkitutils.geom.Vector;
010
011public class VectorArg extends AsbtractSenderArg<Vector> {
012
013    static double parseVectorComponent(CommandsMessages messages, String component) {
014        try {
015            return Double.parseDouble(component);
016        } catch (NumberFormatException exc) {
017            throw new CommandUsageError(
018                    messages.notAValidVectorComponent(component));
019        }
020    }
021
022    public VectorArg(CommandsMessages messages, String string) {
023        super(messages, string);
024    }
025
026    @Override
027    public Vector value() {
028        String[] splitted = string.split(",");
029
030        if (splitted.length > 3) {
031            throw new CommandUsageError();
032        }
033
034        double x = parseVectorComponent(messages, splitted[0]);
035        double y = 0.0;
036        double z = 0.0;
037        if (splitted.length > 1) {
038            y = parseVectorComponent(messages, splitted[1]);
039            if (splitted.length > 2) {
040                z = parseVectorComponent(messages, splitted[2]);
041            }
042        }
043
044        return new Vector(x, y, z);
045    }
046
047    @Override
048    protected Vector defaultFor(CommandSender sender) {
049        if (!(sender instanceof Player)) {
050            throw new CommandUsageError(messages.positionNeeded());
051        }
052
053        return new Vector((Player) sender);
054    }
055}
