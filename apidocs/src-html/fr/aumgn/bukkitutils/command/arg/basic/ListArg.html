001package fr.aumgn.bukkitutils.command.arg.basic;
002
003import static java.util.Collections.singletonList;
004
005import java.util.ArrayList;
006import java.util.Iterator;
007import java.util.List;
008
009import org.bukkit.command.CommandSender;
010
011import com.google.common.collect.AbstractIterator;
012import com.google.common.collect.Iterators;
013
014import fr.aumgn.bukkitutils.command.arg.CommandArg;
015import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
016import fr.aumgn.bukkitutils.command.arg.MatchingArg;
017import fr.aumgn.bukkitutils.command.arg.SenderArg;
018import fr.aumgn.bukkitutils.command.arg.SenderMatchingArg;
019
020public class ListArg<V, T extends CommandArg<V>> implements CommandArg<List<V>>,
021        MatchingArg<V>, SenderArg<List<V>>, SenderMatchingArg<V>, Iterable<T> {
022
023    private final String string;
024    private final String separator;
025    private final CommandArgFactory<T> factory;
026
027    public ListArg(String separator,
028            CommandArgFactory<T> factory, String string) {
029        this.string = string;
030        this.separator = separator;
031        this.factory = factory;
032    }
033
034    @Override
035    public List<V> value() {
036        String[] splitted = string.split(separator);
037        List<V> list = new ArrayList<V>();
038
039        for (String subArg : splitted) {
040            list.add(factory
041                    .createCommandArg(subArg)
042                    .value());
043        }
044
045        return list;
046    }
047
048    public List<V> valueOrSingleton(V def) {
049        return valueOr(singletonList(def));
050    }
051
052    @Override
053    public List<V> valueOr(List<V> def) {
054        if (string != null) {
055            return value();
056        }
057
058        return def;
059    }
060
061    /**
062     * {@inheritDoc}
063     * Note that the {@link CommandArg} type argument of
064     * this ListArg must implement {@link SenderArg} or
065     * this will throws an UnsupportedOperationException.
066     * 
067     * @throws UnsupportedOperationException
068     */
069    @Override
070    public List<V> valueOr(CommandSender sender) {
071        if (string != null) {
072            return match();
073        }
074
075        T arg = factory.createCommandArg(null);
076        if (!(arg instanceof SenderArg)) {
077            throw new UnsupportedOperationException();
078        }
079
080        @SuppressWarnings("unchecked")
081        SenderArg<V> senderArg = (SenderArg<V>) arg;
082        return singletonList(senderArg.valueOr(sender));
083    }
084
085    /**
086     * {@inheritDoc}
087     * Note that the {@link CommandArg} type argument of
088     * this ListArg must implement {@link SenderArg} or
089     * this will throws an UnsupportedOperationException.
090     * 
091     * @throws UnsupportedOperationException
092     */
093    @Override
094    public List<V> valueWithPermOr(String permission, CommandSender sender) {
095        if (string != null) {
096            return match();
097        }
098
099        T arg = factory.createCommandArg(null);
100        if (!(arg instanceof SenderArg)) {
101            throw new UnsupportedOperationException();
102        }
103
104        @SuppressWarnings("unchecked")
105        SenderArg<V> senderArg = (SenderArg<V>) arg;
106        return singletonList(
107                senderArg.valueWithPermOr(permission, sender));
108    }
109
110    /**
111     * {@inheritDoc}
112     * Note that the {@link CommandArg} type argument of
113     * this ListArg must implement {@link MatchingArg} or
114     * this will throws an UnsupportedOperationException.
115     * 
116     * @throws UnsupportedOperationException
117     */
118    @SuppressWarnings("unchecked")
119    @Override
120    public List<V> match() {
121        String[] splitted = string.split(separator);
122        List<V> list = new ArrayList<V>();
123
124        for (String subArg : splitted) {
125            T arg = factory.createCommandArg(subArg);
126            if (!(arg instanceof MatchingArg)) {
127                throw new UnsupportedOperationException();
128            }
129
130            list.addAll(((MatchingArg<V>) arg).match());
131        }
132
133        return list;
134    }
135
136    /**
137     * {@inheritDoc}
138     * Note that the {@link CommandArg} type argument of
139     * this ListArg must implement {@link MatchingArg} or
140     * this will throws an UnsupportedOperationException.
141     * 
142     * @throws UnsupportedOperationException
143     */
144    @Override
145    public List<V> matchOr(V def) {
146        return matchOr(singletonList(def));
147    }
148
149    /**
150     * {@inheritDoc}
151     * Note that the {@link CommandArg} type argument of
152     * this ListArg must implement {@link MatchingArg} or
153     * this will throws an UnsupportedOperationException.
154     * 
155     * @throws UnsupportedOperationException
156     */
157    @Override
158    public List<V> matchOr(List<V> def) {
159        if (string != null) {
160            return match();
161        }
162
163        return def;
164    }
165
166    /**
167     * {@inheritDoc}
168     * Note that the {@link CommandArg} type argument of
169     * this ListArg must implement {@link SenderMatchingArg}
170     * or this will throws an UnsupportedOperationException.
171     * 
172     * @throws UnsupportedOperationException
173     */
174    @Override
175    public List<V> matchOr(CommandSender sender) {
176        if (string != null) {
177            return match();
178        }
179
180        T arg = factory.createCommandArg(null);
181        if (!(arg instanceof SenderMatchingArg)) {
182            throw new UnsupportedOperationException();
183        }
184
185        @SuppressWarnings("unchecked")
186        SenderMatchingArg<V> senderArg = (SenderMatchingArg<V>) arg;
187        return senderArg.matchOr(sender);
188    }
189
190    /**
191     * {@inheritDoc}
192     * Note that the {@link CommandArg} type argument of
193     * this ListArg must implement {@link SenderMatchingArg}
194     * or this will throws an UnsupportedOperationException.
195     * 
196     * @throws UnsupportedOperationException
197     */
198    @Override
199    public List<V> matchWithPermOr(String permission, CommandSender sender) {
200        if (string != null) {
201            return match();
202        }
203
204        T arg = factory.createCommandArg(null);
205        if (!(arg instanceof SenderMatchingArg)) {
206            throw new UnsupportedOperationException();
207        }
208
209        @SuppressWarnings("unchecked")
210        SenderMatchingArg<V> senderArg = (SenderMatchingArg<V>) arg;
211        return senderArg.matchWithPermOr(permission,
212                sender);
213    }
214
215    @Override
216    public Iterator<T> iterator() {
217        if (string == null) {
218            return Iterators.singletonIterator(factory.createCommandArg(null));
219        } else {
220            final String[] splitted = string.split(separator);
221            return new AbstractIterator<T>() {
222                private int i = 0;
223                @Override
224                protected T computeNext() {
225                    if (i == splitted.length) {
226                        endOfData();
227                    }
228                    return factory.createCommandArg(splitted[++i]);
229                }
230            };
231        }
232    }
233}
