001package fr.aumgn.bukkitutils.command.arg.basic;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.Iterator;
006import java.util.List;
007
008import org.bukkit.command.CommandSender;
009
010import com.google.common.collect.AbstractIterator;
011import com.google.common.collect.Iterators;
012
013import fr.aumgn.bukkitutils.command.CommandsMessages;
014import fr.aumgn.bukkitutils.command.arg.CommandArg;
015import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
016import fr.aumgn.bukkitutils.command.arg.MatchingArg;
017import fr.aumgn.bukkitutils.command.arg.SenderMatchingArg;
018
019public class ListArg<V, T extends CommandArg<V>>
020        implements CommandArg<List<V>>, SenderMatchingArg<V>,
021        Iterable<T> {
022
023    private final String string;
024    private final String separator;
025    private final CommandArgFactory<T> factory;
026
027    public ListArg(CommandsMessages messages, String separator,
028            CommandArgFactory<T> factory, String string) {
029        this.string = string;
030        this.separator = separator;
031        this.factory = factory;
032    }
033
034    @Override
035    public List<V> value() {
036        String[] splitted = string.split(separator);
037        List<V> list = new ArrayList<V>();
038
039        for (String subArg : splitted) {
040            list.add(factory
041                    .createCommandArg(subArg)
042                    .value());
043        }
044
045        return list;
046    }
047
048    public List<V> valueOr(V def) {
049        return valueOr(Collections.singletonList(def));
050    }
051
052    @Override
053    public List<V> valueOr(List<V> def) {
054        if (string != null) {
055            return value();
056        }
057
058        return def;
059    }
060
061    @Override
062    public V valueOr(CommandSender sender) {
063        return null;
064    }
065
066    @Override
067    public V valueWithPermOr(String permission, CommandSender sender) {
068        return null;
069    }
070
071    @SuppressWarnings("unchecked")
072    @Override
073    public List<V> match() {
074        String[] splitted = string.split(separator);
075        List<V> list = new ArrayList<V>();
076
077        for (String subArg : splitted) {
078            T arg = factory.createCommandArg(subArg);
079            if (!(arg instanceof MatchingArg)) {
080                throw new UnsupportedOperationException();
081            }
082
083            list.addAll(((MatchingArg<V>) arg).match());
084        }
085
086        return list;
087    }
088
089    @Override
090    public List<V> matchOr(V def) {
091        return matchOr(Collections.singletonList(def));
092    }
093
094    @Override
095    public List<V> matchOr(List<V> def) {
096        if (string != null) {
097            return match();
098        }
099
100        return def;
101    }
102
103    @Override
104    public List<V> matchOr(CommandSender sender) {
105        return null;
106    }
107
108    @Override
109    public List<V> matchWithPermOr(String permission, CommandSender sender) {
110        return null;
111    }
112
113    @Override
114    public Iterator<T> iterator() {
115        if (string == null) {
116            return Iterators.singletonIterator(factory.createCommandArg(null));
117        } else {
118            final String[] splitted = string.split(separator);
119            return new AbstractIterator<T>() {
120                private int i = 0;
121                @Override
122                protected T computeNext() {
123                    if (i == splitted.length) {
124                        endOfData();
125                    }
126                    return factory.createCommandArg(splitted[++i]);
127                }
128            };
129        }
130    }
131}
