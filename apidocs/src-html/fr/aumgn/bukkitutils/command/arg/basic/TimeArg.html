001package fr.aumgn.bukkitutils.command.arg.basic;
002
003import fr.aumgn.bukkitutils.command.CommandsMessages;
004import fr.aumgn.bukkitutils.command.arg.impl.AbstractCommandArg;
005import fr.aumgn.bukkitutils.command.exception.CommandError;
006import fr.aumgn.bukkitutils.util.TimeUtil;
007import fr.aumgn.bukkitutils.util.TimeUtil.UnknownTimeFormatException;
008import fr.aumgn.bukkitutils.util.TimeUtil.UnknownTimePeriodException;
009
010public class TimeArg extends AbstractCommandArg<Integer> {
011
012    public TimeArg(CommandsMessages messages, String string) {
013        super(messages, string);
014    }
015
016    @Override
017    public Integer value() {
018        try {
019            return TimeUtil.parseTime(string);
020        } catch (UnknownTimeFormatException exc) {
021            throw new CommandError(
022                    messages.unknownTimeFormat(exc.getTime()));
023        } catch (UnknownTimePeriodException exc) {
024            throw new CommandError(
025                    messages.unknownTimePeriod(exc.getPeriod()));
026        }
027    }
028}
