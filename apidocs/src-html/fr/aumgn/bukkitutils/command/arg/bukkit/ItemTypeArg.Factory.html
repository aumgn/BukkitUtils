001package fr.aumgn.bukkitutils.command.arg.bukkit;
002
003import org.bukkit.Material;
004
005import fr.aumgn.bukkitutils.command.CommandsMessages;
006import fr.aumgn.bukkitutils.command.arg.CommandArg;
007import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
008import fr.aumgn.bukkitutils.command.exception.CommandError;
009import fr.aumgn.bukkitutils.itemtype.ItemType;
010
011import fr.aumgn.bukkitutils.util.Util;
012
013public class ItemTypeArg extends CommandArg<ItemType> {
014
015    public static class Factory extends CommandArgFactory<ItemType> {
016
017        @Override
018        public ItemTypeArg createCommandArg(CommandsMessages messages, String string) {
019            return new ItemTypeArg(messages, string);
020        }
021    }
022
023    public static class InvalidItemTypeFormat extends CommandError {
024        private static final long serialVersionUID = 1L;
025
026        public InvalidItemTypeFormat(CommandsMessages messages, String given) {
027            super(messages.invalidMaterialAndDataFormat(given));
028        }
029    }
030
031    public ItemTypeArg(CommandsMessages messages, String string) {
032        super(messages, string);
033    }
034
035    @Override
036    public ItemType value() {
037        String[] splitted = string.split(":");
038        if (splitted.length > 2) {
039            throw new InvalidItemTypeFormat(messages, string);
040        }
041
042        Material material = MaterialArg.getMaterial(messages, splitted[0]);
043        Short data = 0;
044        if (splitted.length == 2) {
045            data = Util.parseDataFor(material, splitted[1]);
046            if (data == null) {
047                throw new InvalidItemTypeFormat(messages, string);
048            }
049        }
050
051        return new ItemType(material, data);
052    }
053}
