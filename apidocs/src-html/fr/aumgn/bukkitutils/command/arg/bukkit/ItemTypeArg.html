001package fr.aumgn.bukkitutils.command.arg.bukkit;
002
003import org.bukkit.Material;
004import org.bukkit.command.CommandSender;
005import org.bukkit.entity.Player;
006
007import fr.aumgn.bukkitutils.command.CommandsMessages;
008import fr.aumgn.bukkitutils.command.arg.impl.AsbtractSenderArg;
009import fr.aumgn.bukkitutils.command.exception.CommandError;
010import fr.aumgn.bukkitutils.itemtype.ItemType;
011
012import fr.aumgn.bukkitutils.util.Util;
013
014public class ItemTypeArg extends AsbtractSenderArg<ItemType> {
015
016    public static class InvalidItemTypeFormat extends CommandError {
017        private static final long serialVersionUID = 1L;
018
019        public InvalidItemTypeFormat(CommandsMessages messages, String given) {
020            super(messages.invalidMaterialAndDataFormat(given));
021        }
022    }
023
024    public ItemTypeArg(CommandsMessages messages, String string) {
025        super(messages, string);
026    }
027
028    @Override
029    public ItemType value() {
030        String[] splitted = string.split(":");
031        if (splitted.length > 2) {
032            throw new InvalidItemTypeFormat(messages, string);
033        }
034
035        Material material = MaterialArg.getMaterial(messages, splitted[0]);
036        Short data = 0;
037        if (splitted.length == 2) {
038            data = Util.parseDataFor(material, splitted[1]);
039            if (data == null) {
040                throw new InvalidItemTypeFormat(messages, string);
041            }
042        }
043
044        return new ItemType(material, data);
045    }
046
047    @Override
048    protected ItemType defaultFor(CommandSender sender) {
049        if (!(sender instanceof Player)) {
050            throw new RuntimeException();
051        }
052
053        return new ItemType(((Player) sender).getItemInHand());
054    }
055}
