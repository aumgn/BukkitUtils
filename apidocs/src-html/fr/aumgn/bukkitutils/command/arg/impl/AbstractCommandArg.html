001package fr.aumgn.bukkitutils.command.arg.impl;
002
003import fr.aumgn.bukkitutils.command.CommandsMessages;
004import fr.aumgn.bukkitutils.command.arg.CommandArg;
005
006public abstract class AbstractCommandArg<V> implements CommandArg<V> {
007
008    protected final CommandsMessages messages;
009    protected final String string;
010
011    public AbstractCommandArg(CommandsMessages messages, String string) {
012        this.messages = messages;
013        this.string = string;
014    }
015
016    @Override
017    public V valueOr(V def) {
018        if (string == null) {
019            return def;
020        }
021
022        return value();
023    }
024}
