001package fr.aumgn.bukkitutils.command.arg;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007import org.bukkit.command.CommandSender;
008
009import fr.aumgn.bukkitutils.command.CommandsMessages;
010
011public class CommandListArg<V> extends AbstractCommandArg<List<V>, V> {
012
013    private final String separator;
014    private final CommandArgFactory<V> factory;
015
016    public CommandListArg(CommandsMessages messages,
017            CommandArgFactory<V> factory, String string) {
018        this(messages, ",", factory, string);
019    }
020
021    public CommandListArg(CommandsMessages messages, String separator,
022            CommandArgFactory<V> factory, String string) {
023        super(messages, string);
024        this.separator = separator;
025        this.factory = factory;
026    }
027
028    @Override
029    public List<V> value() {
030        String[] splitted = string.split(separator);
031        List<V> list = new ArrayList<V>();
032
033        for (String subArg : splitted) {
034            list.add(factory
035                    .createCommandArg(messages, subArg)
036                    .value());
037        }
038
039        return list;
040    }
041
042    @Override
043    public List<V> valueOr(List<V> def) {
044        return super.valueOr(def);
045    }
046
047    public List<V> valueOrSingleton(V def) {
048        return valueOr(Collections.<V>singletonList(def));
049    }
050
051    @Override
052    protected List<V> defaultFor(CommandSender sender) {
053        V value = factory
054                .createCommandArg(messages, null)
055                .defaultFor(sender);
056        return Collections.<V>singletonList(value);
057    }
058
059    @Override
060    public List<V> match() {
061        String[] splitted = string.split(separator);
062        List<V> list = new ArrayList<V>();
063
064        for (String subArg : splitted) {
065            list.addAll(factory
066                    .createCommandArg(messages, subArg)
067                    .match());
068        }
069
070        return list;
071    }
072
073    @Override
074    public List<V> matchOr(List<V> def) {
075        return super.matchOr(def);
076    }
077
078    @Override
079    public List<V> matchOr(V def) {
080        return matchOr(Collections.<V>singletonList(def));
081    }
082
083    @Override
084    protected List<V> defaultMatchFor(CommandSender sender) {
085        return factory
086                .createCommandArg(messages, null)
087                .defaultMatchFor(sender);
088    }
089}
